62 void _do_add_to(
63  chaiscript::ChaiScript& chai,
64  metaobject auto mo,
65  metaobject auto ms) {
66  if constexpr(reflects_object_sequence(mo)) {
67  for_each(mo, [&](auto me) { _do_add_to(chai, me, ms); });
68  } else if constexpr(reflects_base(mo)) {
69  _add_base_class(
70  chai, get_reflected_type(get_class(mo)), get_reflected_type(ms));
71  } else {
72  const std::string name{get_name(mo)};
73  if constexpr(reflects_variable(mo)) {
74  if constexpr(reflects_record_member(mo)) {
75  if constexpr(is_public(mo)) {
76  chai.add(chaiscript::fun(get_pointer(mo)), name);
77  }
78  } else {
79  chai.add(chaiscript::var(get_reference(mo)), name);
80  }
81  } else if constexpr(reflects_constructor(mo)) {
82  if constexpr(is_public(mo)) {
83  _add_constructor(
84  chai,
85  get_reflected_type(get_scope(mo)),
86  extract_types(transform(get_parameters(mo), get_type(_1))),
87  name);
88  }
89  } else if constexpr(reflects_function(mo)) {
90  if constexpr(reflects_record_member(mo) && is_public(mo)) {
91  if constexpr(reflects_conversion_operator(mo)) {
92  if constexpr(!is_deleted(mo)) {
93  _add_conversion(
94  chai,
95  get_reflected_type(get_scope(mo)),
96  get_reflected_type(get_type(mo)));
97  }
98  } else {
99  if constexpr(!is_deleted(mo)) {
100  chai.add(chaiscript::fun(get_pointer(mo)), name);
101  }
102  }
103  } else {
104  chai.add(chaiscript::fun(get_pointer(mo)), name);
105  }
106  } else if constexpr(reflects_record(mo)) {
107  _add_type(chai, get_reflected_type(mo), name);
108  _do_add_to(chai, get_base_classes(mo), mo);
109  _do_add_to(chai, get_member_types(mo), mo);
110  _do_add_to(chai, get_data_members(mo), mo);
111  _do_add_to(chai, get_constructors(mo), mo);
112  _do_add_to(chai, get_member_functions(mo), mo);
113  _do_add_to(chai, get_operators(mo), mo);
114  }
115  }
116 }

118 static inline void add_to(chaiscript::ChaiScript& chai, metaobject auto mo) {
119  _do_add_to(chai, mo, no_metaobject);
120 }
