18 class puml_class_diagram {
19 private:
20  void _add_access_marker(std::ostream& out, const metadata& md) const {
21  if(md.has(trait::is_private)) {
22  out << "-";
23  } else if(md.has(trait::is_protected)) {
24  out << "#";
25  } else {
26  out << "+";
27  }
28  }
29 
30  void _add_parameter_list(std::ostream& out, const metadata& md) {
31  out << "(";
32  bool first = true;
33  for(const auto& mp : md.parameters()) {
34  if(first) {
35  first = false;
36  } else {
37  out << ", ";
38  }
39  out << mp.type().name_();
40  }
41  out << ")";
42  }
43 
44  void _gen_enum_decl(std::ostream& out, const metadata& md) {
45  out << "enum ";
46  out << md.name_() << " {\n";
47  for(const auto& me : md.enumerators()) {
48  out << " " << me.name_() << '\n';
49  }
50  out << "}\n";
51  }
52 
53  void _gen_record_decl(std::ostream& out, const metadata& md) {
54  if(md.has(trait::uses_class_key)) {
55  out << "class ";
56  } else if(md.has(trait::uses_struct_key)) {
57  out << "class ";
58  } else {
59  out << "union ";
60  }
61  out << md.name_() << " {\n";
62  bool separate_members = false;
63  for(const auto& mm : md.data_members()) {
64  if(mm.has(trait::is_public)) {
65  out << " ";
66  _add_access_marker(out, mm);
67  if(mm.has(trait::is_static)) {
68  out << "{static}";
69  }
70  out << mm.name_();
71  out << ": ";
72  out << mm.type().name_();
73  out << "\n";
74  separate_members = true;
75  }
76  }
77  bool separate_constructors = false;
78  for(const auto& mc : md.constructors()) {
79  if(mc.has_none(
80  trait::is_copy_constructor | trait::is_move_constructor)) {
81  if(separate_members) {
82  out << " __\n";
83  separate_members = false;
84  }
85  out << " ";
86  _add_access_marker(out, mc);
87  out << mc.name_();
88  _add_parameter_list(out, mc);
89  out << "\n";
90  separate_constructors = true;
91  }
92  }
93  for(const auto& mf : md.member_functions().with_name()) {
94  if(separate_members) {
95  out << " __\n";
96  separate_members = false;
97  }
98  if(separate_constructors) {
99  out << " ..\n";
100  separate_constructors = false;
101  }
102  out << " ";
103  _add_access_marker(out, mf);
104  if(mf.has(trait::is_pure_virtual)) {
105  out << "{abstract}";
106  }
107  out << mf.type().name_() << " ";
108  out << mf.name_();
109  _add_parameter_list(out, mf);
110  out << "\n";
111  }
112  out << "}\n";
113  }
114 
115  void _gen_decl(std::ostream& out, const metadata& md) {
116  if(md.has(trait::reflects_enum)) {
117  _gen_enum_decl(out, md);
118  } else if(md.has(trait::reflects_record)) {
119  _gen_record_decl(out, md);
120  }
121  }
122 
123  void _gen_rel(
124  std::ostream& out,
125  const metadata& md,
126  const metadata_sequence& ms) {
127  if(md.has(trait::reflects_class)) {
128  for(const auto& mb : md.base_classes()) {
129  if(ms.contains(mb.class_())) {
130  out << mb.class_().name_();
131  if(mb.has(trait::is_virtual)) {
132  out << " <|.. ";
133  } else {
134  out << " <|-- ";
135  }
136  out << md.name_() << '\n';
137  }
138  }
139  }
140  if(md.has(trait::reflects_record)) {
141  std::set<metadata_value> owns;
142  std::set<metadata_value> refs;
143  std::set<metadata_value> uses;
144 
145  for(const auto& mm : md.data_members()) {
146  if(mm.base_type().element_type()) {
147  const auto& mt = mm.base_type().element_type().base_type();
148  if(ms.contains(mt)) {
149  if(mt.has(trait::is_pointer | trait::is_reference)) {
150  refs.emplace(mt);
151  } else {
152  owns.emplace(mt);
153  }
154  }
155  } else {
156  const auto& mt = mm.base_type();
157  if(ms.contains(mt)) {
158  if(mt.has(trait::is_pointer | trait::is_reference)) {
159  refs.emplace(mt);
160  } else {
161  uses.emplace(mt);
162  }
163  }
164  }
165  }
166  for(const auto& mf : md.constructors() + md.member_functions()) {
167  if(ms.contains(mf.base_type())) {
168  uses.emplace(mf.base_type());
169  }
170  for(const auto& mp : mf.parameters()) {
171  if(ms.contains(mp.base_type())) {
172  uses.emplace(mp.base_type());
173  }
174  }
175  }
176  for(const auto& mv : uses) {
177  if(
178  (mv != md) && !owns.contains(mv) && !refs.contains(mv) &&
179  mv->has_none(trait::is_fundamental)) {
180  out << md.name_() << " --> " << mv->name_() << '\n';
181  }
182  }
183  for(const auto& mv : owns) {
184  if((mv != md) && mv->has_none(trait::is_fundamental)) {
185  out << md.name_() << " *-- " << mv->name_() << '\n';
186  }
187  }
188  for(const auto& mv : refs) {
189  if(mv->has_none(trait::is_fundamental)) {
190  out << md.name_() << " o-- " << mv->name_() << '\n';
191  }
192  }
193  }
194  }
195 
196 public:
197  std::ostream& generate(std::ostream& out, const metadata_sequence& ms) {
198  const auto named = ms.with_name().not_having(
199  type_trait::is_const | type_trait::is_volatile | trait::is_pointer |
200  trait::is_reference);
201 
202  out << "@startuml\n";
203  for(const auto& md : named) {
204  _gen_decl(out, md);
205  }
206  out << '\n';
207  for(const auto& md : named) {
208  _gen_rel(out, md, named);
209  }
210  out << "@enduml\n";
211  return out;
212  }
213 };
