17 template <typename T>

22 template <__metaobject_id M, typename C, typename A, size_t... I>
23 constexpr auto do_apply_on(
24  wrapped_metaobject<M> mo,
25  C& obj,
26  A&& args,
27  std::index_sequence<I...>) requires(__metaobject_is_meta_member_function(M)) {
28  if constexpr(is_static(mo)) {
29  return (*_get_pointer<M>::value)(std::get<I>(std::forward<A>(args))...);
30  } else {
31  return (obj.*_get_pointer<M>::value)(
32  std::get<I>(std::forward<A>(args))...);
33  }
34 }
