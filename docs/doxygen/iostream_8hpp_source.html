48  auto
49  select_constructor(construction_context_param ctx, const factory& fac)
50  -> size_t {
51  size_t pick = 0U;
52  const auto bad_pick = [](const auto& ctr) {
53  return ctr.is_copy_constructor() || ctr.is_move_constructor();
54  };
55 
56  while(true) {
57  ctx.out << "select constructor:" << std::endl;
58  const auto ctr_count = fac.constructor_count();
59  for(size_t c = 0; c < ctr_count; ++c) {
60  const auto& ctr = fac.constructor(c);
61  if(!bad_pick(ctr)) {
62  ctx.out << " " << c << ": " << fac.product_type_name()
63  << "(";
64  const auto param_count = ctr.parameter_count();
65  for(size_t p = 0; p < param_count; ++p) {
66  if(p) {
67  ctx.out << ", ";
68  }
69  ctx.out << ctr.parameter(p).type_name();
70  ctx.out << " ";
71  ctx.out << ctr.parameter(p).name();
72  }
73  ctx.out << ")" << std::endl;
74  }
75  }
76  ctx.in >> pick;
77  if(pick < ctr_count) {
78  if(!bad_pick(fac.constructor(pick))) {
79  break;
80  }
81  }
82  }
83  return pick;
84  }
85  };

99  static auto make_path_of(const factory_constructor_parameter& param) {
100  auto path = std::string(param.name());
101  auto* pparam = &param;
102  while(true) {
103  if(!(pparam = pparam->parent_parameter())) {
104  break;
105  }
106  path = std::string(pparam->name()) + "." + path;
107  }
108  return path;
109  }
