90  template <typename R, typename T>
91  static constexpr auto
92  _do_get_referenced_type(std::type_identity<T>, R& r) noexcept
93  -> const metadata& {
94  if constexpr(std::is_pointer_v<T>) {
95  using P =
96  std::remove_cv_t<typename std::pointer_traits<T>::value_type>;
97  return get_metadata(get_aliased(mirror(P)), r);
98  } else if constexpr(std::is_reference_v<T>) {
99  using P = std::remove_cv_t<std::remove_reference_t<T>>;
100  return get_metadata(get_aliased(mirror(P)), r);
101  } else {
102  return r.get_none();
103  }
104  }

139  template <operation_metaobject O, __metaobject_id M>
140  static auto _try_init(
141  wrapped_metaobject<M> mo,
142  metadata_registry& r,
143  const metadata*& pmd) -> const metadata* {
144  if(pmd->is_none()) {
145  if constexpr(mirror::is_applicable<O>(mo)) {
146  const auto ms = mirror::try_apply<O>(mo);
147  auto& md = get_metadata(ms, r);
148  pmd = &md;
149  if constexpr(!reflects_type(ms)) {
150  md.init(ms, r);
151  }
152  }
153  }
154  return pmd;
155  }

297 /// @brief Class storing run-time metadata objects.
298 /// @ingroup metadata
299 class metadata_registry {
300 private:
301  std::map<hash_t, std::unique_ptr<stored_metadata>> _metadata;
302 
303  friend auto get_no_metadata(metadata_registry& r) noexcept
304  -> const metadata& {
305  return r.get_none();
306  }
307 
308  template <__metaobject_id M>
309  auto _get(wrapped_metaobject<M> mo) noexcept -> stored_metadata& {
310  const auto id = get_hash(mo);
311  auto pos = _metadata.find(id);
312  if(pos == _metadata.end()) {
313  pos = _metadata
314  .emplace(
315  id,
316  std::make_unique<stored_metadata>(
317  mo, id, get_full_name(mo), *this))
318  .first;
319  }
320  return *pos->second;
321  }
322 
323  template <__metaobject_id M>
324  friend auto
325  get_metadata(wrapped_metaobject<M> mo, metadata_registry& r) noexcept
326  -> stored_metadata& {
327  return r._get(mo);
328  }
329 
330  auto _find(metaobject auto mo) -> stored_metadata& {
331  const auto id = get_hash(mo);
332  auto pos = _metadata.find(id);
333  if(pos == _metadata.end()) {
334  throw metadata_not_found();
335  }
336  return *pos->second;
337  }
338 
339  template <__metaobject_id M>
340  auto _add(wrapped_metaobject<M> mo) noexcept -> stored_metadata& {
341  auto& md = _get(mo);
342  md.init(mo, *this);
343  return md;
344  }
345 
346 public:
347  /// @brief Default constructor.
348  metadata_registry() noexcept {
349  _metadata.emplace(
350  get_hash(no_metaobject), std::make_unique<stored_metadata>());
351  }
352 
353  /// @brief Returns the count of the contained metadata objects.
354  /// @see all
355  auto size() const noexcept {
356  return _metadata.size();
357  }
358 
359  /// @brief Returns an iterator to the first metaobject.
360  /// @see end
361  /// @see all
362  /// @see size
363  auto begin() const noexcept -> metadata_registry_iterator {
364  return {_metadata.begin()};
365  }
366 
367  /// @brief Returns an iterator past the last metaobject.
368  /// @see begin
369  /// @see all
370  /// @see size
371  auto end() const noexcept -> metadata_registry_iterator {
372  return {_metadata.end()};
373  }
374 
375  /// @brief Returns the special no-metadata object.
376  auto get_none() noexcept -> const metadata& {
377  return *_metadata[get_hash(no_metaobject)];
378  }
379 
380  /// @brief Registers and stores the metadata from a metaobject.
381  /// @see find
382  /// @see all
383  auto add(metaobject auto mo) noexcept -> const metadata& {
384  return _add(mo);
385  }
386 
387  /// @brief Tries to find the metadata matching to the specified metaobject.
388  /// @see add
389  /// @throws metadata_not_found
390  auto find(metaobject auto mo) -> const metadata& {
391  return _find(mo);
392  }
393 
394  /// @brief Returns a sequence of all stored metadata instances.
395  /// @see filtered
396  /// @see begin
397  /// @see end
398  /// @see size
399  auto all() const -> metadata_sequence {
400  std::vector<const metadata*> elements;
401  elements.reserve(_metadata.size());
402  for(const auto& p : _metadata) {
403  elements.push_back(std::get<1>(p).get());
404  }
405  return {elements};
406  }
407 
408  /// @brief Returns a sequence of stored metadata instances satisfying a predicate.
409  /// @see all
410  /// @see begin
411  /// @see end
412  /// @see size
413  template <typename F>
414  auto filtered(F predicate) const -> metadata_sequence {
415  std::vector<const metadata*> elements;
416  elements.reserve(_metadata.size());
417  for(const auto& p : _metadata) {
418  if(predicate(*std::get<1>(p))) {
419  elements.push_back(std::get<1>(p).get());
420  }
421  }
422  return {elements};
423  }
