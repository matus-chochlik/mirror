75  template <typename T>
76  auto write(const write_driver& drv, context_param ctx, const T& value)
77  -> write_errors {
78  if constexpr(std::is_same_v<T, bool>) {
79  ctx.node.SetBool(value);
80  } else if constexpr(std::is_same_v<T, tribool>) {
81  if(value) {
82  ctx.node.SetBool(true);
83  } else if(!value) {
84  ctx.node.SetBool(false);
85  } else {
86  ctx.node.SetNull();
87  }
88  } else if constexpr(std::is_integral_v<T>) {
89  if constexpr(std::is_signed_v<T>) {
90  if constexpr(sizeof(T) > 4Z) {
91  ctx.node.SetInt64(value);
92  } else {
93  ctx.node.SetInt(value);
94  }
95  } else {
96  if constexpr(sizeof(T) > 4Z) {
97  ctx.node.SetUint64(value);
98  } else {
99  ctx.node.SetUint(value);
100  }
101  }
102  } else if constexpr(std::is_floating_point_v<T>) {
103  if constexpr(std::is_same_v<T, float>) {
104  ctx.node.SetFloat(value);
105  } else {
106  ctx.node.SetDouble(value);
107  }
108  } else if constexpr(std::is_convertible_v<T, std::string_view>) {
109  const std::string_view view{value};
110  ctx.node.SetString(to_rapidjson(view));
111  } else {
112  return drv.write(*this, ctx, value);
113  }
114  return {};
115  }
