17 /// @brief Class for manipulating and testing a group of enumeration-based bits.
18 /// @ingroup utilities
19 template <typename Bit>
20 class bitfield {
21 public:
22  /// @brief The enumeration type specifying individual bit values.
23  using bit_type = Bit;
24 
25  /// @brief The Integral type used to store the bits.
26  using value_type = std::make_unsigned_t<std::underlying_type_t<bit_type>>;
27 
28  /// @brief Default constructor.
29  constexpr bitfield() noexcept = default;
30 
31  /// @brief Explicit construction from the underlying interger type value.
32  constexpr explicit bitfield(const value_type bits) noexcept
33  : _bits{bits} {}
34 
35  /// @brief Construction from the bit enumeration type value.
36  constexpr bitfield(const bit_type _bit) noexcept
37  : _bits{value_type(_bit)} {}
38 
39  constexpr bitfield(const bit_type _bit_a, const bit_type _bit_b) noexcept
40  : _bits{value_type(_bit_a) | value_type(_bit_b)} {}
41 
42  constexpr bitfield(std::initializer_list<bit_type> bits) noexcept {
43  for(const auto bit : bits) {
44  (*this) |= bit;
45  }
46  }
47 
48  /// @brief Indicates that none of the bits are set.
49  constexpr auto is_empty() const noexcept {
50  return _bits == value_type(0);
51  }
52 
53  /// @brief Indicates that some of the bits are set.
54  /// @see is_empty
55  explicit constexpr operator bool() const noexcept {
56  return _bits != value_type(0);
57  }
58 
59  /// @brief Explicit conversion to the underlying integer value type.
60  /// @see bits
61  explicit constexpr operator value_type() const noexcept {
62  return _bits;
63  }
64 
65  /// @brief Returns the bits in the underlying integer value type.
66  constexpr auto bits() const noexcept -> value_type {
67  return _bits;
68  }
69 
70  /// @brief Tests if the specified bit is set.
71  /// @see has_not
72  /// @see has_only
73  /// @see has_all
74  /// @see has_any
75  /// @see has_none
76  /// @see has_at_most
77  constexpr auto has(const bit_type bit) const noexcept {
78  return (_bits & value_type(bit)) == value_type(bit);
79  }
80 
81  /// @brief Tests if the specified bit is not set.
82  /// @see has
83  /// @see has_all
84  /// @see has_any
85  /// @see has_only
86  /// @see has_none
87  /// @see has_at_most
88  constexpr auto has_not(const bit_type bit) const noexcept {
89  return (_bits & value_type(bit)) != value_type(bit);
90  }
91 
92  /// @brief Tests if all of the specified bits are set.
93  /// @see has
94  /// @see has_not
95  /// @see has_only
96  /// @see has_any
97  /// @see has_none
98  /// @see has_at_most
99  template <typename... B>
100  constexpr auto has_all(const bit_type bit, B... bits) const noexcept
101  -> bool {
102  return (has(bit) && ... && has(bits));
103  }
104 
105  /// @brief Tests if all of the specified bits are set.
106  /// @see has
107  /// @see has_not
108  /// @see has_only
109  /// @see has_any
110  /// @see has_none
111  /// @see has_at_most
112  constexpr auto has_all(const bitfield that) const noexcept {
113  return (_bits & that._bits) == that._bits;
114  }
115 
116  /// @brief Tests if any of the specified bits are set.
117  /// @see has
118  /// @see has_not
119  /// @see has_only
120  /// @see has_some
121  /// @see has_all
122  /// @see has_none
123  /// @see has_at_most
124  template <typename... B>
125  constexpr auto has_any(const bit_type bit, B... bits) const noexcept
126  -> bool {
127  return (has(bit) || ... || has(bits));
128  }
129 
130  /// @brief Tests if some of the specified bits are set.
131  /// @see has
132  /// @see has_not
133  /// @see has_any
134  /// @see has_none
135  /// @see has_at_most
136  constexpr auto has_some(const bitfield that) const noexcept {
137  return (_bits & that._bits) != value_type{0};
138  }
139 
140  /// @brief Tests if none of the specified bits are set.
141  /// @see has
142  /// @see has_not
143  /// @see has_only
144  /// @see has_all
145  /// @see has_any
146  /// @see has_at_most
147  template <typename... B>
148  constexpr auto has_none(const bit_type bit, B... bits) const noexcept
149  -> bool {
150  return (has_not(bit) && ... && has_not(bits));
151  }
152 
153  /// @brief Tests if none of the specified bits are set.
154  /// @see has
155  /// @see has_not
156  /// @see has_any
157  /// @see has_some
158  /// @see has_at_most
159  constexpr auto has_none(const bitfield that) const noexcept {
160  return (_bits & that._bits) == value_type{0};
161  }
162 
163  /// @brief Tests if only the specified bit is set.
164  /// @see has
165  /// @see has_not
166  /// @see has_all
167  /// @see has_any
168  /// @see has_none
169  /// @see has_at_most
170  constexpr auto has_only(const bit_type bit) const noexcept {
171  return _bits == value_type(bit);
172  }
173 
174  /// @brief Tests if at most the specified bit is set (or is empty).
175  /// @see has
176  /// @see has_not
177  /// @see has_only
178  /// @see has_all
179  /// @see has_any
180  /// @see has_none
181  constexpr auto has_at_most(const bit_type bit) const noexcept {
182  return is_empty() || has_only(bit);
183  }
184 
185  /// @brief Equality comparison.
186  friend constexpr auto
187  operator==(const bitfield a, const bitfield b) noexcept {
188  return a._bits == b._bits;
189  }
190 
191  /// @brief Nonequality comparison.
192  friend constexpr auto
193  operator!=(const bitfield a, const bitfield b) noexcept {
194  return a._bits != b._bits;
195  }
196 
197  /// @brief Bitwise-or operator.
198  friend constexpr auto operator|(const bitfield a, const bitfield b) noexcept
199  -> bitfield {
200  return bitfield(a._bits | b._bits);
201  }
202 
203  /// @brief Bitwise-or operator.
204  auto operator|=(const bitfield b) noexcept -> bitfield& {
205  _bits |= b._bits;
206  return *this;
207  }
208 
209  /// @brief Bitwise-and operator.
210  friend constexpr auto operator&(const bitfield a, const bitfield b) noexcept
211  -> bitfield {
212  return bitfield(a._bits & b._bits);
213  }
214 
215  /// @brief Bitwise-and operator.
216  auto operator&=(const bitfield b) noexcept -> bitfield& {
217  _bits &= b._bits;
218  return *this;
219  }
220 
221  /// @brief Bit inversion operator
222  friend constexpr auto operator~(const bitfield b) noexcept -> bitfield {
223  return bitfield{value_type(~b._bits)};
224  }
225 
226  /// @brief Clears the specified bit.
227  auto clear(const bit_type b) noexcept -> bitfield& {
228  _bits &= ~value_type(b); // NOLINT(hicpp-signed-bitwise)
229  return *this;
230  }
231 
232  /// @brief Clears all bits.
233  /// @post is_empty()
234  auto clear() noexcept -> bitfield& {
235  _bits = value_type(0);
236  return *this;
237  }
238 
239 private:
240  value_type _bits{0U};
241 };
