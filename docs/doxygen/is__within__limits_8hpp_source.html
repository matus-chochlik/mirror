28 struct implicitly_within_limits
29  : std::integral_constant<
30  bool,
31  (((std::is_integral_v<Dst> && std::is_integral_v<Src>) ||
32  (std::is_floating_point_v<Dst> && std::is_floating_point_v<Src>)) &&
33  (std::is_signed_v<Dst> ==
34  std::is_signed_v<Src>)&&(sizeof(Dst) >= sizeof(Src)))> {};

66 //------------------------------------------------------------------------------
67 template <typename Dst, typename Src, bool IsInt, bool IsSig>
68 struct within_limits_num<Dst, Src, IsInt, IsInt, IsSig, IsSig> {
69  static constexpr auto check(const Src value) noexcept {
70  using dnl = std::numeric_limits<Dst>;
71 
72  return (dnl::min() <= value) && (value <= dnl::max());
73  }
74 };
75 //------------------------------------------------------------------------------
76 template <typename Dst, typename Src, bool IsInt>
77 struct within_limits_num<Dst, Src, IsInt, IsInt, false, true> {
78  static constexpr auto check(const Src value) noexcept {
79  using Dnl = std::numeric_limits<Dst>;
80  using Tmp = std::make_unsigned_t<Src>;
81 
82  return (value < Src(0)) ? false : (Tmp(value) < Dnl::max());
83  }
84 };

87 struct within_limits_num<Dst, Src, IsInt, IsInt, true, false> {
88  static constexpr auto check(const Src value) noexcept {
89  using dnl = std::numeric_limits<Dst>;

111 //------------------------------------------------------------------------------
112 /// @brief Indicates if @p value fits into the specified Dst type.
113 /// @ingroup utilities
114 /// @see limit_cast
115 /// @see convert_if_fits
116 ///
117 /// This function tests if the specified argument would fit into another type.
118 /// For example if a value stored in 64-bit integer can be converted into
119 /// a 16-bit integer without overflow.
120 template <typename Dst, typename Src>
121 static constexpr auto is_within_limits(const Src value) noexcept {
122  return implicitly_within_limits<Dst, Src>::value ||
123  within_limits<Dst, Src>::check(value);
124 }
125 //------------------------------------------------------------------------------
126 /// @brief Casts @p value to Dst type if the value fits in that type.
127 /// @ingroup utilities
128 /// @see is_within_limits
129 /// @see convert_if_fits
130 /// @see signedness_cast
131 /// @pre is_within_limits<Dst>(value)
132 template <typename Dst, typename Src>
133 static constexpr auto limit_cast(Src value) noexcept
134  -> std::enable_if_t<std::is_convertible_v<Src, Dst>, Dst> {
135  is_within_limits<Dst>(value), Dst(std::move(value));
136 }
137 //------------------------------------------------------------------------------
138 /// @brief Casts @p value to a type with the opposite signedness.
139 /// @ingroup utilities
140 /// @see is_within_limits
141 /// @see convert_if_fits
142 /// @see limit_cast
143 template <typename Src>
144 static constexpr auto signedness_cast(Src value) noexcept {
145  return limit_cast<std::conditional_t<
146  std::is_signed_v<Src>,
147  std::make_unsigned_t<Src>,
148  std::make_signed_t<Src>>>(value);
149 }
150 //------------------------------------------------------------------------------
151 /// @brief Optionally converts @p value to Dst type if the value fits in that type.
152 /// @ingroup utilities
153 /// @see is_within_limits
154 /// @see limit_cast
155 template <typename Dst, typename Src>
156 static constexpr auto convert_if_fits(Src value) noexcept
157  -> std::enable_if_t<std::is_convertible_v<Src, Dst>, std::optional<Dst>> {
158 
159  if(is_within_limits<Dst>(value)) {
160  return {Dst(std::move(value))};
161  }
162  return {};
163 }
