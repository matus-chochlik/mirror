162 inline auto url::_get_regex() noexcept -> const std::regex& {
163 #ifdef __clang__
164  MIRROR_DIAG_PUSH()
165  MIRROR_DIAG_OFF(exit-time-destructors)
166 #endif
167  // clang-format off
168  static const std::regex re{
169  R"(^((([\w]+):)?\/\/)(([^:]+)(:(\S+))?@)?((((\w[\w_-]{0,62}(\.\w[\w_-]{0,62})*))|((10|127)(\.\d{1,3}){3})|((169\.254|192\.168)(\.\d{1,3}){2})|(172\.(1[6-9]|2\d|3[0-1])(\.\d{1,3}){2})|([1-9]\d?|1\d\d|2[01]\d|22[0-3])(\.(1?\d{1,2}|2[0-4]\d|25[0-5])){2}(\.([1-9]\d?|1\d\d|2[0-4]\d|25[0-4])))(:(\d{1,5}))?)?((/[\w_-]+)*/?)?(\?(([\w_]+=[^+#]*)(\+([\w_]+=[^+#]*))*))?(#([\w_-]*))?$)",
170  std::regex::ECMAScript};
171  // clang-format on
172 #ifdef __clang__
173  MIRROR_DIAG_POP()
174 #endif
175  return re;
176 }

178 inline void url::_cover(
179  std::string_view& part,
180  const std::match_results<std::string::iterator>& match,
181  const std::size_t index) const noexcept {
182  assert(index < match.size());
183  auto& m = match[index];
184  part = str().substr(
185  static_cast<std::size_t>(m.first - _url_str.begin()),
186  static_cast<std::size_t>(m.second - m.first));
187 }
188 //------------------------------------------------------------------------------
189 inline url::url(
190  std::string url_str,
191  std::match_results<std::string::iterator> match) noexcept
192  : _url_str{std::move(url_str)}
193  , _parsed{
194  std::regex_match(_url_str.begin(), _url_str.end(), match, _get_regex())} {
195 
196  if(_parsed) {
197  _cover(_scheme, match, 3);
198  _cover(_login, match, 5);
199  _cover(_passwd, match, 7);
200  _cover(_host, match, 9);
201  _cover(_port, match, 28);
202  _cover(_path, match, 29);
203  _cover(_query, match, 32);
204  _cover(_fragment, match, 37);
205  }
206 }
207 //------------------------------------------------------------------------------
208 auto url::argument(const std::string_view arg_name) const noexcept
209  -> std::optional<std::string_view> {
210  auto src = _query;
211  while(!src.empty()) {
212  const auto pos = src.find('+');
213  const auto pair = src.substr(0Z, pos);
214  if(pair.starts_with(arg_name)) {
215  return pair.substr(arg_name.size() + 1Z);
216  }
217  src = src.substr(pos + 1Z);
218  }
219  return {};
220 }
