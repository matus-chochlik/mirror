23 /// @brief Concept constraining extractable types.

49 /// @brief Returns the value type of an extractable.
50 /// @ingroup utilities
51 /// @see extract
52 template <typename T>
53 using extracted_type_t = std::remove_cv_t<typename extracted_traits<
54  std::remove_cv_t<std::remove_reference_t<T>>>::value_type>;
55 
56 template <typename E, typename V>
57 static constexpr const auto has_value_type_v =
58  std::is_convertible_v<extracted_type_t<E>, V>;

179 template <typename T>
180 concept extractable = requires(T v) {
181  { has_value(v) } -> std::convertible_to<bool>;
182  { std::declval<mirror::extracted_type_t<T>>() };
