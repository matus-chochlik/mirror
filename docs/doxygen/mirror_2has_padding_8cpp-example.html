template <typename... T>
constexpr auto sum_sizeofs(mirror::type_list<T...>) {
return (0Z + ... + sizeof(T));
}
template <typename T>
constexpr auto has_padding() -> bool {
return sizeof(T) >
sum_sizeofs(extract_types(transform(
filter(get_data_members(mirror(T)), not_(is_static(mirror::_1))),
get_type(mirror::_1))));
}
template <typename T>
void print_has_padding() {
std::cout << get_name(remove_all_aliases(mirror(T))) << ": "
<< (has_padding<T>() ? "has some" : "has no") << " padding"
<< std::endl;
}
int main() {
print_has_padding<S1>();
print_has_padding<S2>();
return 0;
}
