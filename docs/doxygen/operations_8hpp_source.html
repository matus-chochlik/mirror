172  /// @brief Returns a sequence of member function reflections of a reflected class.
173  get_member_functions = 1ULL << 8ULL,
174  /// @brief Returns a sequence of member type reflections of a reflected class.
175  get_member_types = 1ULL << 9ULL,
176  /// @brief Returns a sequence of operator reflections of a reflected class.
177  get_operators = 1ULL << 10ULL,
178  /// @brief Returns a sequence of parameter reflections of a reflected callable.
179  get_parameters = 1ULL << 11ULL,
180  /// @brief Returns a sequence of enumerator reflections of a reflected enum.
181  get_enumerators = 1ULL << 12ULL,
182  /// @brief Returns the reflection of the scope in a reflected entity.
183  get_scope = 1ULL << 13ULL,
184  /// @brief Returns a reflection of the scope of a reflected scope member.
185  get_subexpression = 1ULL << 14ULL,
186  /// @brief Returns a reflection of the type of a reflected typed entity.
187  get_type = 1ULL << 15ULL,
188  /// @brief Returns a reflection of the underlying type of a reflected enum,
189  get_underlying_type = 1ULL << 16ULL,
